# 电话号码的字母组合

##### 思路：

通过设置StringBuilder  combin ， 把combin 作为往结果List中的一个元素 ， 每一个完整循环生成一个combin ， 在递归达到边界时 往List中添加combin

##### 方法： 回溯法

```java
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

class Solution {
    private final Map<Character, String> map = new HashMap<Character, String>() {
        {
            put('2', "abc");
            put('3', "def");
            put('4', "ghi");
            put('5', "jkl");
            put('6', "mno");
            put('7', "pqrs");
            put('8', "tuv");
            put('9', "wxyz");
        }
    };

    List<String> list = new ArrayList<>();

    public List<String> letterCombinations(String digits) {
        if (digits == null || digits.length() == 0) {
            return list;
        }
        StringBuilder str = new StringBuilder();
        backtracking(0, str, digits);
        return list;
    }

    public void backtracking(int index, StringBuilder combin, String digits) {
        if (index == digits.length()) {
            list.add(combin.toString());
            return;
        }
        char digit = digits.charAt(index);
        String letters = map.get(digit);
        for (char letter : letters.toCharArray()) {
            combin.append(letter);
            backtracking(index + 1, combin, digits);
            combin.deleteCharAt(combin.length() - 1);
        }
    }

}
```
